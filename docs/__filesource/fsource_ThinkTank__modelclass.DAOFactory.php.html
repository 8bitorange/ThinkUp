  2. /**
  3. * DAOFactory
  4. * 
  5. * inits a dao based on the think tank config db_type and $dao_mapping definitions.
  6.  
  7. * db_type is defined in webapp/config.inc.php as:
  8. * 
  9. *     $THINKTANK_CFG['db_type'] = 'somedb';
  10. * 
  11. * Example: DAOFactory::getDAO('SomeDAO');

  13. * @author Mark Wilkie
  14. */
  15. class DAOFactory {
  16.  
  17.     /**
  18.      * maps DAO from db_type and defines class names and path for initialization
  19.      */
  20.     static $dao_mapping = array (
  21.         // our test dao
  22.         'TestDAO' => array( 
  23.             'mysql' => array( 'class' => 'TestMysqlDAO', 'path' =>  'tests/classes/class.TestMysqlDAO.php'),
  24.             'faux' => array( 'class' => 'TestFauxDAO', 'path' =>  'tests/classes/class.TestFauxDAO.php'),
  25.         ),
  26.         //Instance MySQL DAO
  27.         'InstanceDAO' => array(
  28.             'mysql' => array( 'class' => 'InstanceMySQLDAO', 'path' => 'model/class.InstanceMySQLDAO.php')
  29.         )
  30.     );
  31.  
  32.     /*
  33.      *  Creates a DAO instance and returns it
  34.      *
  35.      * @param string - the name of the dao you wish to init
  36.      * @returns object - a concrete dao instance
  37.      */
  38.     public static function getDAO($dao_key) {
  39.         $db_type = self::getDBType();
  40.         if(! isset(self::$dao_mapping[$dao_key]) ) {
  41.             throw new Exception("No DAO mapping defined for: " . $dao_key);
  42.         }
  43.         if(! isset(self::$dao_mapping[$dao_key][$db_type])) {
  44.             throw new Exception("No db mapping defined for '" . $dao_key . "' with db type: " . $db_type);
  45.         }
  46.         $class_info = self::$dao_mapping[$dao_key][$db_type];
  47.         require_once($class_info['path']);
  48.         $dao = new $class_info['class'];
  49.         return $dao;
  50.     }
  51.  
  52.     /**
  53.      * gets the db_type for our configured ThinkTank instance, defaults to mysql,
  54.      * db_type can optionally be defined in webapp/config.inc.php as:
  55.      * 
  56.      *     $THINKTANK_CFG['db_type'] = 'somedb';
  57.      * 
  58.      * @return string db_type, will default to 'mysql' if not defined
  59.      */
  60.     public static function getDBType() {
  61.         $type = Config::getInstance()->getValue('db_type');
  62.         $type = is_null($type) ? 'mysql' : $type;
  63.         return $type;
  64.     }
  65. }
