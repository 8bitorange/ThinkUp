  104.     function facebook_process_page_actions() {
  105.         global $db;
  106.         $messages = array("error"=>'', "success"=>'', "info"=>'');

  127.             }
  128.             $tokens = $oid->getOAuthTokens($existing_instance_id);
  129.             $session_key = $tokens['oauth_access_token'];
  130.             $oid->insert($owner_id, $instance_id, $session_key);
  131.         } else {
  132.             $messages["info"] .= "Instance ".$fb_page_id.", facebook exists.";
  133.             $instance_id = $i->id;
