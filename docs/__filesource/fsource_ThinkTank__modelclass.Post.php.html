  602.         $post = $this->getPost($orphan_id);
  603.

  606.         if ($former_parent_id < 0 && isset($post->in_reply_to_post_id) && $this->isPostInDB($post->in_reply_to_post_id)) {

  610.         $q = "
  611.             UPDATE 
  612.                 #prefix#posts
  613.             SET 
  614.                 in_reply_to_post_id = ".$parent_id."
  615.             WHERE
  616.                 post_id = ".$orphan_id;
  617.         $this->executeSQL($q);

  619.  
  620.         if ($parent_id > 0) {
