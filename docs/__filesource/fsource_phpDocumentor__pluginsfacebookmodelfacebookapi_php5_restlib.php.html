  200.   private function execute_server_side_batch($read_only) {
  201.     $item_count = count($this->batch_queue);
  202.     $method_feed = array();
  203.     foreach ($this->batch_queue as $batch_item) {
  204.       $method = $batch_item['m'];
  205.       $params = $batch_item['p'];
  206.       list($get, $post) = $this->finalize_params($method, $params);
  207.       $method_feed[] = $this->create_url_string(array_merge($post, $get));
  208.     }
  209.  
  210.     $serial_only =
  211.       ($this->batch_mode == FacebookRestClient::BATCH_MODE_SERIAL_ONLY);
  212.  
  213.     $params = array('method_feed' => json_encode($method_feed),
  214.                     'serial_only' => $serial_only,
  215.                     'format' => $this->format);
  216.     $result = $this->call_method('facebook.batch.run', $params, $read_only);
  217.  
  218.     if (is_array($result) && isset($result['error_code'])) {
  219.       throw new FacebookRestClientException($result['error_msg'],
  220.                                             $result['error_code']);
  221.     }
  222.  
  223.     for ($i = 0; $i < $item_count; $i++) {
  224.       $batch_item = $this->batch_queue[$i];
  225.       $batch_item['p']['format'] = $this->format;
  226.       $batch_item_result = $this->convert_result($result[$i],
  227.                                                  $batch_item['m'],
  228.                                                  $batch_item['p']);
  229.  
  230.       if (is_array($batch_item_result) &&
  231.           isset($batch_item_result['error_code'])) {
  232.         throw new FacebookRestClientException($batch_item_result['error_msg'],
  233.                                               $batch_item_result['error_code']);
  234.       }
  235.       $batch_item['r'] = $batch_item_result;
  236.     }
  237.   }
