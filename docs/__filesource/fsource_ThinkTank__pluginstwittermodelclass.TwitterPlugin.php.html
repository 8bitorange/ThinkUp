  24.             $noauth = true;
  25.  
  26.             if (isset($tokens['oauth_access_token']) && $tokens['oauth_access_token'] != '' && isset($tokens['oauth_access_token_secret']) && $tokens['oauth_access_token_secret'] != '') {

  29.  
  30.             if ($noauth) {
  31.                 $api = new CrawlerTwitterAPIAccessorOAuth('NOAUTH', 'NOAUTH', $config->getValue('oauth_consumer_key'), $config->getValue('oauth_consumer_secret'), $i, $config->getValue('archive_limit'));
  32.             } else {
  33.                 $api = new CrawlerTwitterAPIAccessorOAuth($tokens['oauth_access_token'], $tokens['oauth_access_token_secret'], $config->getValue('oauth_consumer_key'), $config->getValue('oauth_consumer_secret'), $i, $config->getValue('archive_limit'));
  34.             }
  35.

  37.  
  38.             $api->init();
  39.

  54.                     $crawler->fetchInstanceUserFriends();

  60.                 $crawler->fetchFriendTweetsAndFriends();
  61.  
  62.                 // TODO: Get direct messages
  63.                 // TODO: Gather favorites data
  64.  
  65.                 if ($noauth) {
  66.                     // No auth req'd
  67.                     $crawler->fetchSearchResults($i->network_username);
  68.                 }
  69.  
  70.                 $crawler->cleanUpFollows();
  71.  
  72.                 // Save instance
  73.                 $id->save($crawler->instance, $crawler->owner_object->post_count, $logger, $api);

  107.                     $s->assign('errormsg', $e->getMessage());
  108.                 }
  109.                 if (isset($user) && $user["is_protected"] == 'false') {
  110.                     // if so, add to instances table and owners table
  111.  
  112.                     $i = $id->getByUsername($_GET['twitter_username']);
  113.                     $oid = new OwnerInstanceDAO($db);
  114.  
  115.                     $msg = '';
  116.                     if (isset($i)) {
  117.                         //$msg .= "Instance already exists.<br />";
  118.  
  119.                         $oi = $oid->get($owner->id, $i->id);
  120.                         if ($oi != null) {
  121.                             //$msg .= "Owner already has this instance, no insert or update required.<br />";
  122.                         } else {
  123.                             $oid->insert($owner->id, $i->id, '', '');
  124.                             //$msg .= "Added owner instance.<br />";
  125.                         }
  126.  
  127.                     } else {
  128.                         //$msg .= "Instance does not exist.<br />";
  129.  
  130.                         $id->insert($user["user_id"], $user["user_name"]);
  131.                         //$msg .= "Created instance.<br />";
  132.  
  133.                         $i = $id->getByUsername($user["user_name"]);
  134.                         $oid->insert($owner->id, $i->id, '', '');
  135.                         //$msg .= "Created an owner instance.<br />";
  136.                     }
  137.                     $s->assign('successmsg', $_GET['twitter_username']." has been added to ThinkTank.");
  138.  
  139.                     $s->assign('successmsg', "Added ".$_GET['twitter_username']." to ThinkTank.");
  140.                 } else { // if not, return error
  141.                     $s->assign('errormsg', $_GET['twitter_username']." is a private Twitter account; ThinkTank cannot track it without authorization.");
  142.                 }
  143.             } else {
  144.                 $s->assign('errormsg', $_GET['twitter_username']." is not a valid Twitter username.");
  145.             }
  146.         }
  147.  
  148.         $to = new TwitterOAuth($oauth_consumer_key, $oauth_consumer_secret);
  149.         /* Request tokens from twitter */
  150.         $tok = $to->getRequestToken();
  151.         if (isset($tok['oauth_token'])) {
  152.             $token = $tok['oauth_token'];
  153.             $_SESSION['oauth_request_token_secret'] = $tok['oauth_token_secret'];
  154.  
  155.             /* Build the authorization URL */
  156.             $oauthorize_link = $to->getAuthorizeURL($token);
  157.         } else {
  158.             //set error message here
  159.             $s->assign('errormsg', "Unable to obtain OAuth token. Check your Twitter consumer key and secret configuration.");
  160.             $oauthorize_link = '';
  161.         }
  162.  
  163.         $owner_instances = $id->getByOwnerAndNetwork($owner, 'twitter');
  164.  
  165.         $s->assign('owner_instances', $owner_instances);
  166.         $s->assign('oauthorize_link', $oauthorize_link);
