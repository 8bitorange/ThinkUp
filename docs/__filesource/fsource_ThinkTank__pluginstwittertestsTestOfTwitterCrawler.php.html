  169.     function testFetchRetweetsOfInstanceuser() {
  170.         self::setUpInstanceUserGinaTrapani();
  171.         $tc = new TwitterCrawler($this->instance, $this->api, $this->db);
  172.         $tc->fetchInstanceUserInfo();
  173.  
  174.         //first, load retweeted tweet into db
  175.         $q = "INSERT INTO tt_posts (post_id, author_user_id, author_username, author_fullname, author_avatar, post_text, source, pub_date, mention_count_cache, retweet_count_cache) VALUES (14947487415, 930061, 'ginatrapani', 'Gina Trapani', 'avatar.jpg', '&quot;Wearing your new conference tee shirt does NOT count as dressing up.&quot;', 'web', '2006-01-01 00:00:00', ".rand(0, 4).", 0);";
  176.         $this->db->exec($q);
  177.  
  178.         $pdao = new PostDAO($this->db);
  179.         $tc->fetchRetweetsOfInstanceUser();
  180.         $post = $pdao->getPost(14947487415);
  181.         $this->assertEqual($post->retweet_count_cache, 6, '6 retweets loaded');
  182.         $retweets = $pdao->getRetweetsOfPost(14947487415, true);
  183.         $this->assertEqual(sizeof($retweets), 6, '6 retweets loaded');
  184.  
  185.         //make sure duplicate posts aren't going into the db on next crawler run
  186.         self::setUpInstanceUserGinaTrapani();
  187.         $tc = new TwitterCrawler($this->instance, $this->api, $this->db);
  188.         $tc->fetchInstanceUserInfo();
  189.  
  190.         $tc->fetchRetweetsOfInstanceUser();
  191.         $post = $pdao->getPost(14947487415);
  192.         $this->assertEqual($post->retweet_count_cache, 6, '6 retweets loaded');
  193.         $retweets = $pdao->getRetweetsOfPost(14947487415, true);
  194.         $this->assertEqual(sizeof($retweets), 6, '6 retweets loaded');
