  29.     function untinyurl($tinyurl, $ldao) {

  41.  
  42.         if (!isset($url['path'])) {
  43.             $logger->logstatus("$tinyurl has no path", "Expand URLs Plugin");

  47.             $url = $url['path'].$query.$fragment;
  48.             $request = "HEAD {$url} HTTP/1.0\r\nHost: {$host}\r\nConnection: Close\r\n\r\n";
  49.  
  50.             fwrite($sock, $request);

  58.                     list(, $location) = explode(':', $line, 2);
  59.                     return ltrim($location);
  60.                 }
  61.             }
  62.             return $tinyurl;
  63.         }
